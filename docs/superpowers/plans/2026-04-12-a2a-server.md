**Goal:** Add A2A protocol support to generated AgentStack servers — Agent Card discovery endpoint, JSON-RPC task handling, streaming, and interrupt/resume support.

**Architecture:** New `a2a.py` module in the adapter generates A2A-specific code (Agent Card, TaskManager, JSON-RPC handlers). The `templates.py` `generate_server_py` function calls into `a2a.py` to append A2A endpoints to the existing server. All A2A handlers delegate to the same LangGraph agent as the REST API.

**Tech Stack:** Python 3.11+, FastAPI, JSON-RPC 2.0, SSE (sse-starlette), pytest

```python
"""A2A protocol code generation for AgentStack servers."""

from agentstack.schema.agent import Agent


def generate_agent_card_code(agent: Agent) -> str:
    """Generate Agent Card constant and endpoint code."""
    description = (agent.instructions or agent.name)[:500].replace('"', '\\"')

    skills = []
    for skill in agent.skills:
        tool_tags = ", ".join(f'"{t}"' for t in skill.tools)
        tool_desc = ", ".join(skill.tools) if skill.tools else "general"
        skills.append(
            f'    {{"id": "{skill.name}", "name": "{skill.name}", '
            f'"description": "Tools: {tool_desc}", "tags": [{tool_tags}]}}'
        )
    skills_str = ",\n".join(skills) if skills else ""

    lines = []
    lines.append("")
    lines.append("# === A2A Agent Card ===")
    lines.append(f'AGENT_CARD = {{')
    lines.append(f'    "name": "{agent.name}",')
    lines.append(f'    "description": "{description}",')
    lines.append(f'    "url": "http://agentstack-{agent.name}:8000",')
    lines.append(f'    "version": "0.1.0",')
    lines.append(f'    "capabilities": {{"streaming": True, "pushNotifications": False}},')
    if skills_str:
        lines.append(f'    "skills": [')
        lines.append(skills_str)
        lines.append(f'    ],')
    else:
        lines.append(f'    "skills": [],')
    lines.append(f'    "defaultInputModes": ["text/plain"],')
    lines.append(f'    "defaultOutputModes": ["text/plain"],')
    lines.append(f'}}')
    lines.append("")
    lines.append("")
    lines.append('@app.get("/.well-known/agent.json")')
    lines.append("async def agent_card():")
    lines.append("    return AGENT_CARD")
    lines.append("")

    return "\n".join(lines)


def generate_task_manager_code() -> str:
    """Generate TaskManager class code."""
    return '''
# === A2A Task Manager ===
class TaskManager:
    """Tracks A2A task states."""

    def __init__(self):
        self._tasks = {}

    async def create_task(self, task_id: str) -> dict:
        from datetime import datetime, timezone
        task = {
            "id": task_id,
            "state": "submitted",
            "created_at": datetime.now(timezone.utc).isoformat(),
            "updated_at": datetime.now(timezone.utc).isoformat(),
            "history": [],
            "last_message": None,
        }
        self._tasks[task_id] = task
        return task

    async def get_task(self, task_id: str) -> dict | None:
        return self._tasks.get(task_id)

    async def update_task(self, task_id: str, state: str, message: dict | None = None) -> dict:
        from datetime import datetime, timezone
        task = self._tasks.get(task_id)
        if task is None:
            task = await self.create_task(task_id)
        task["state"] = state
        task["updated_at"] = datetime.now(timezone.utc).isoformat()
        if message:
            task["last_message"] = message
            task["history"].append(message)
        return task

_task_manager = None
'''


def generate_a2a_handler_code(agent: Agent) -> str:
    """Generate A2A JSON-RPC handler code."""
    agent_ref = "_agent" if True else "agent"  # always use _agent pattern for A2A

    return f'''
# === A2A JSON-RPC Handler ===
from fastapi import Request
from langgraph.types import Command


def _jsonrpc_response(req_id, result):
    return {{"jsonrpc": "2.0", "id": req_id, "result": result}}


def _jsonrpc_error(req_id, code, message):
    return {{"jsonrpc": "2.0", "id": req_id, "error": {{"code": code, "message": message}}}}


def _extract_text(message):
    """Extract text from A2A message parts."""
    parts = message.get("parts", [])
    texts = []
    for part in parts:
        if "text" in part:
            texts.append(part["text"])
    return " ".join(texts) if texts else ""


def _make_a2a_message(role, text):
    """Create an A2A message."""
    return {{"role": role, "parts": [{{"text": text}}]}}


def _extract_response_text(result):
    """Extract text from LangGraph result."""
    content = result["messages"][-1].content
    if isinstance(content, list):
        return "".join(
            block.get("text", "") if isinstance(block, dict) else str(block)
            for block in content
        )
    return str(content)


@app.post("/a2a")
async def a2a_handler(request: Request):
    body = await request.json()
    req_id = body.get("id")
    method = body.get("method")
    params = body.get("params", {{}})

    if method == "tasks/send":
        return await _handle_tasks_send(req_id, params)
    elif method == "tasks/get":
        return await _handle_tasks_get(req_id, params)
    elif method == "tasks/cancel":
        return await _handle_tasks_cancel(req_id, params)
    elif method == "tasks/sendSubscribe":
        return EventSourceResponse(_handle_tasks_stream(req_id, params))
    else:
        return _jsonrpc_error(req_id, -32601, f"Method not found: {{method}}")


async def _handle_tasks_send(req_id, params):
    task_id = params.get("id", str(uuid.uuid4()))
    message = params.get("message", {{}})
    text = _extract_text(message)

    # Get or create task
    task = await _task_manager.get_task(task_id)

    if task and task["state"] == "input_required":
        # Resume from interrupt
        await _task_manager.update_task(task_id, "working")
        result = await {agent_ref}.ainvoke(
            Command(resume=text),
            config={{"configurable": {{"thread_id": task_id}}}},
        )
    else:
        # New task or continued conversation
        if not task:
            await _task_manager.create_task(task_id)
        await _task_manager.update_task(
            task_id, "working",
            _make_a2a_message("user", text),
        )
        result = await {agent_ref}.ainvoke(
            {{"messages": [("user", text)]}},
            config={{"configurable": {{"thread_id": task_id}}}},
        )

    # Check for interrupt
    if "__interrupt__" in result and result["__interrupt__"]:
        interrupt_value = result["__interrupt__"][0].value if hasattr(result["__interrupt__"][0], "value") else str(result["__interrupt__"][0])
        interrupt_text = str(interrupt_value) if not isinstance(interrupt_value, str) else interrupt_value
        agent_msg = _make_a2a_message("agent", interrupt_text)
        await _task_manager.update_task(task_id, "input_required", agent_msg)
        task = await _task_manager.get_task(task_id)
        return _jsonrpc_response(req_id, {{
            "id": task_id,
            "status": {{"state": "input_required", "message": agent_msg, "timestamp": task["updated_at"]}},
        }})

    # Completed
    response_text = _extract_response_text(result)
    agent_msg = _make_a2a_message("agent", response_text)
    await _task_manager.update_task(task_id, "completed", agent_msg)
    task = await _task_manager.get_task(task_id)
    return _jsonrpc_response(req_id, {{
        "id": task_id,
        "status": {{"state": "completed", "message": agent_msg, "timestamp": task["updated_at"]}},
    }})


async def _handle_tasks_get(req_id, params):
    task_id = params.get("id")
    if not task_id:
        return _jsonrpc_error(req_id, -32602, "Missing task id")
    task = await _task_manager.get_task(task_id)
    if not task:
        return _jsonrpc_error(req_id, -32001, f"Task not found: {{task_id}}")
    return _jsonrpc_response(req_id, {{
        "id": task_id,
        "status": {{"state": task["state"], "timestamp": task["updated_at"]}},
        "history": task["history"],
    }})


async def _handle_tasks_cancel(req_id, params):
    task_id = params.get("id")
    if not task_id:
        return _jsonrpc_error(req_id, -32602, "Missing task id")
    task = await _task_manager.get_task(task_id)
    if not task:
        return _jsonrpc_error(req_id, -32001, f"Task not found: {{task_id}}")
    await _task_manager.update_task(task_id, "canceled")
    task = await _task_manager.get_task(task_id)
    return _jsonrpc_response(req_id, {{
        "id": task_id,
        "status": {{"state": "canceled", "timestamp": task["updated_at"]}},
    }})


async def _handle_tasks_stream(req_id, params):
    task_id = params.get("id", str(uuid.uuid4()))
    message = params.get("message", {{}})
    text = _extract_text(message)

    task = await _task_manager.get_task(task_id)
    if not task:
        await _task_manager.create_task(task_id)

    await _task_manager.update_task(task_id, "working", _make_a2a_message("user", text))

    # Send working status
    yield {{"data": json.dumps({{
        "id": task_id,
        "status": {{"state": "working"}},
        "final": False,
    }})}}

    # Stream tokens
    async for event in {agent_ref}.astream_events(
        {{"messages": [("user", text)]}},
        config={{"configurable": {{"thread_id": task_id}}}},
        version="v2",
    ):
        if event["event"] == "on_chat_model_stream":
            token = event["data"]["chunk"].content
            if token:
                yield {{"data": json.dumps({{
                    "id": task_id,
                    "artifact": {{"parts": [{{"text": token}}]}},
                    "append": True,
                }})}}

    # Final status
    response_text = ""
    try:
        result = await {agent_ref}.aget_state(
            config={{"configurable": {{"thread_id": task_id}}}},
        )
        if result.values.get("messages"):
            content = result.values["messages"][-1].content
            if isinstance(content, list):
                response_text = "".join(
                    block.get("text", "") if isinstance(block, dict) else str(block)
                    for block in content
                )
            else:
                response_text = str(content)
    except Exception:
        pass

    agent_msg = _make_a2a_message("agent", response_text)
    await _task_manager.update_task(task_id, "completed", agent_msg)
    yield {{"data": json.dumps({{
        "id": task_id,
        "status": {{"state": "completed", "message": agent_msg}},
        "final": True,
    }})}}
'''
```
